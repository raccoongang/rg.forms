How Conditional Attributes Work


    
disabled_when

    
max_users = ReactiveIntegerField(
    disabled_when="$subscription_type == 'free'",
)
# Renders: data-attr:disabled="$subscription_type == 'free'"


    
read_only_when

    
api_key = ReactiveCharField(
    read_only_when="$subscription_type != 'premium'",
)
# Renders: data-attr:readonly="$subscription_type != 'premium'"


    
help_text_when

    
max_users = ReactiveIntegerField(
    help_text_when={
        "$subscription_type == 'free'": "Upgrade to customize",
        "$subscription_type == 'basic'": "Up to 10 users",
        "$subscription_type == 'premium'": "Unlimited users",
    },
)
# Renders multiple help texts with data-show for each condition
